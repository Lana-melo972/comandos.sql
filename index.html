Comandos SQL Explicados com Exemplos

DDL – Data Definition Language

Usado para definir ou modificar a estrutura do banco de dados.

CREATE DATABASE

Cria um novo banco de dados.

CREATE DATABASE Escola;

USE

Seleciona qual banco será utilizado.

USE Escola;

CREATE TABLE

Cria uma nova tabela no banco.

CREATE TABLE Alunos (
    id INT PRIMARY KEY,
    nome VARCHAR(100),
    idade INT
);

ALTER TABLE

Modifica a estrutura de uma tabela existente.

ALTER TABLE Alunos ADD email VARCHAR(100);

DROP COLUMN

Remove uma coluna de uma tabela.

ALTER TABLE Alunos DROP COLUMN email;

DESCRIBE

Exibe a estrutura de uma tabela.

DESCRIBE Alunos;

DML – Data Manipulation Language

Usado para manipular os dados nas tabelas.

INSERT INTO

Insere dados em uma tabela.

INSERT INTO Alunos (id, nome, idade) VALUES (1, 'Maria', 17);

UPDATE

Atualiza dados existentes.

UPDATE Alunos SET idade = 18 WHERE id = 1;

DELETE FROM

Remove dados de uma tabela.

DELETE FROM Alunos WHERE id = 1;

Cláusulas de Filtro

Usadas para filtrar os resultados das consultas.

WHERE

Aplica uma condição para filtrar os dados.

SELECT * FROM Alunos WHERE idade > 17;

AND, OR, NOT

Combinam condições lógicas.

SELECT * FROM Alunos WHERE idade > 17 AND nome = 'Maria';
SELECT * FROM Alunos WHERE NOT idade < 18;

LIKE

Filtra valores com base em padrões.

SELECT * FROM Alunos WHERE nome LIKE 'M%'; -- Começa com M
SELECT * FROM Alunos WHERE nome LIKE '%a'; -- Termina com a

Consultas e Relacionamentos

Para consultar dados e unir tabelas relacionadas.

SELECT

Seleciona dados de uma tabela.

SELECT nome, idade FROM Alunos;

JOIN

Relaciona dados de duas tabelas com base em uma chave.

SELECT Alunos.nome, Notas.nota
FROM Alunos
JOIN Notas ON Alunos.id = Notas.aluno_id;
